# B1008 数组元素循环右移问题
## tag 模拟 
### 题意 给定n个数 一个整数m 使得这n个数循环右移m个单位 且不开辟额外空间
### 解法 三次反转
### 代码
```cpp
#include <bits/stdc++.h>
using namespace std;
using gg = long long;
int main()
{
   ios::sync_with_stdio(false);
   cin.tie(0);
   int n,m;
   cin>>n>>m;
   vector<int> v(n);
   for(int i=0;i<n;i++){
       cin>>v[i];
   }
   m= m % n ; 
   /*for(int i=n-m;i<=n-1;i++) {
       if(i != n-m) cout<<" ";
       cout<<v[i];
   }
   if(m) cout<<" ";
   for(int i=0;i<=n-m-1;i++){
       if(i) cout<<" ";
       cout<<v[i];
   }*/
   // 考研数据结构题目
   reverse(v.begin()+n-m,v.end());
   reverse(v.begin(),v.begin()+n-m); // 这里参数应该是n-m 因为其右侧取不到
   reverse(v.begin(),v.end());
   for(int i=0;i<n;i++){
       if(i) cout<<" ";
       cout<<v[i];
   }
   return 0;
}
```
